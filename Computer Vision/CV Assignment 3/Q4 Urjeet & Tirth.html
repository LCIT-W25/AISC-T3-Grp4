# Cell 5: Define the Diffusion Model

# Text Encoder using BERT
class TextEncoder(nn.Module):
    def __init__(self, output_dim=256):  # Increased output_dim for richer text embeddings
        super().__init__()
        self.bert = BertModel.from_pretrained('bert-base-uncased')
        self.fc = nn.Linear(768, output_dim)  # Reduce BERT output to save memory
        # Freeze BERT weights to save memory and speed up training
        for param in self.bert.parameters():
            param.requires_grad = False
    
    def forward(self, input_ids, attention_mask):
        with torch.no_grad():  # No gradients for BERT (frozen)
            outputs = self.bert(input_ids=input_ids, attention_mask=attention_mask)
        pooled_output = outputs.pooler_output  # [batch_size, 768]
        return self.fc(pooled_output)  # [batch_size, output_dim]

# Simplified U-Net with Gradient Checkpointing
class UNet(nn.Module):
    def __init__(self, in_channels=3, out_channels=3, text_dim=256, base_channels=64):  # Increased text_dim and base_channels
        super().__init__()
        self.time_dim = 128  # Increased time embedding dimension for better capacity
        
        # Time embedding
        self.time_mlp = nn.Sequential(
            nn.Linear(1, self.time_dim),
            nn.ReLU(),
            nn.Linear(self.time_dim, self.time_dim)
        )
        
        # Initial convolution
        self.conv0 = nn.Conv2d(in_channels, base_channels, 3, padding=1)
        self.bn0 = nn.BatchNorm2d(base_channels)  # Added batch norm for stability
        
        # Downsampling (convert to ModuleList for checkpointing)
        self.down1 = nn.ModuleList([
            nn.Conv2d(base_channels, base_channels * 2, 3, stride=2, padding=1),
            nn.BatchNorm2d(base_channels * 2),
            nn.ReLU(),
            nn.Conv2d(base_channels * 2, base_channels * 2, 3, padding=1),
            nn.BatchNorm2d(base_channels * 2),
            nn.ReLU()
        ])
        
        # Time embedding projection to match downsampled channels
        self.time_proj = nn.Linear(self.time_dim, base_channels * 2)
        
        # Text embedding integration
        self.text_mlp = nn.Linear(text_dim, base_channels * 2)
        
        # Upsampling (convert to ModuleList for checkpointing)
        self.up1 = nn.ModuleList([
            nn.ConvTranspose2d(base_channels * 2, base_channels, 4, stride=2, padding=1),
            nn.BatchNorm2d(base_channels),
            nn.ReLU(),
            nn.Conv2d(base_channels, base_channels, 3, padding=1),
            nn.BatchNorm2d(base_channels),
            nn.ReLU()
        ])
        
        # Output
        self.out = nn.Conv2d(base_channels, out_channels, 3, padding=1)
    
    def forward(self, x, t, text_emb):
        # Time embedding
        t = t.unsqueeze(-1).float()
        t_emb = self.time_mlp(t)  # [batch_size, time_dim]
        
        # Initial conv
        x = self.conv0(x)  # [batch_size, base_channels, 96, 96] (updated for new image size)
        x = self.bn0(x)
        x = F.relu(x)
        
        # Downsampling with checkpointing
        x = checkpoint_sequential(self.down1, 2, x)  
        
        # Add time embedding
        t_emb = self.time_proj(t_emb)  
        t_emb = t_emb.view(-1, t_emb.size(1), 1, 1)  
        t_emb = t_emb.repeat(1, 1, x.size(2), x.size(3))  
        x = x + t_emb
        
        # Add text embedding
        text_emb = self.text_mlp(text_emb)  
        text_emb = text_emb.view(-1, text_emb.size(1), 1, 1)  
        text_emb = text_emb.repeat(1, 1, x.size(2), x.size(3))  
        x = x + text_emb
        
        # Upsampling with checkpointing
        x = checkpoint_sequential(self.up1, 2, x)  
        
        return self.out(x)  

# Diffusion Model
class DiffusionModel(nn.Module):
    def __init__(self, base_channels=64):  # Increased base_channels
        super().__init__()
        self.text_encoder = TextEncoder(output_dim=256)  # Matches updated text_dim
        self.unet = UNet(base_channels=base_channels)  # Allow base_channels to be configurable
        self.beta = torch.linspace(1e-4, 0.02, 1000).to(device)  # Noise schedule
        self.alpha = 1.0 - self.beta
        self.alpha_bar = torch.cumprod(self.alpha, dim=0)
    
    def forward(self, x, t, input_ids, attention_mask):
        text_emb = self.text_encoder(input_ids, attention_mask)
        return self.unet(x, t, text_emb)
    
    def add_noise(self, x, t):
        noise = torch.randn_like(x).to(device)
        alpha_bar_t = self.alpha_bar[t].view(-1, 1, 1, 1)
        noisy_x = torch.sqrt(alpha_bar_t) * x + torch.sqrt(1 - alpha_bar_t) * noise
        return noisy_x, noise
    
    def sample(self, input_ids, attention_mask, num_steps=500, shape=(8, 3, 96, 96)):  
        self.eval()
        # Ensure batch size matches input_ids
        if input_ids.size(0) != shape[0]:
            raise ValueError(f"Batch size mismatch: input_ids has batch size {input_ids.size(0)}, but shape expects {shape[0]}")
        with torch.no_grad():
            x = torch.randn(shape).to(device)
            for t in reversed(range(num_steps)):
                t_tensor = torch.full((shape[0],), t, device=device, dtype=torch.long)
                with autocast():
                    noise_pred = self(x, t_tensor, input_ids, attention_mask)
                alpha_t = self.alpha[t]
                alpha_bar_t = self.alpha_bar[t]
                beta_t = self.beta[t]
                x = (1 / torch.sqrt(alpha_t)) * (x - ((1 - alpha_t) / torch.sqrt(1 - alpha_bar_t)) * noise_pred)
                if t > 0:
                    z = torch.randn_like(x).to(device)
                    x += torch.sqrt(beta_t) * z
            # Clamp output to match normalized image range [-1, 1]
            x = torch.clamp(x, -1.0, 1.0)
        return x

# Initialize model and optimizer with optimized parameters
model = DiffusionModel(base_channels=64).to(device)  # Updated base_channels

# Function to preprocess images for Inception V3
def preprocess_for_inception(images):
    # Images are in range [-1, 1], denormalize to [0, 1] using training stats
    images = images * torch.tensor([0.229, 0.224, 0.225], device=device).view(1, 3, 1, 1) + torch.tensor([0.485, 0.456, 0.406], device=device).view(1, 3, 1, 1)
    # Clamp to ensure valid range [0, 1]
    images = torch.clamp(images, 0, 1)
    # Resize to 299x299 as required by Inception V3
    images = F.interpolate(images, size=(299, 299), mode='bilinear', align_corners=False)
    # Normalize to ImageNet stats (Inception V3 expects this)
    mean = torch.tensor([0.485, 0.456, 0.406]).view(1, 3, 1, 1).to(device)
    std = torch.tensor([0.229, 0.224, 0.225]).view(1, 3, 1, 1).to(device)
    images = (images - mean) / std
    return images

# Function to compute Inception Score (IS)
def compute_inception_score(images, num_splits=10):
    with torch.no_grad():
        preds = []
        for i in range(0, len(images), batch_size):
            batch = images[i:i+batch_size].to(device)
            batch = preprocess_for_inception(batch)
            pred = inception_model(batch)
            pred = F.softmax(pred, dim=1)
            preds.append(pred.cpu().numpy())
            torch.cuda.empty_cache()  # Clear memory after each batch
        preds = np.concatenate(preds, axis=0)
    
    # Compute IS
    scores = []
    split_size = len(preds) // num_splits
    for k in range(num_splits):
        part = preds[k * split_size:(k + 1) * split_size]
        py = np.mean(part, axis=0)
        kl_div = part * (np.log(part + 1e-8) - np.log(py + 1e-8))
        kl_div = np.mean(np.sum(kl_div, axis=1))
        scores.append(np.exp(kl_div))
    return np.mean(scores), np.std(scores)

            batch = preprocess_for_inception(batch)
            feat = inception_model(batch)
            gen_features.append(feat.cpu().numpy())
            torch.cuda.empty_cache()  # Clear memory after each batch
        gen_features = np.concatenate(gen_features, axis=0)
    
    # Compute mean and covariance
    mu1, sigma1 = np.mean(real_features, axis=0), np.cov(real_features, rowvar=False)
    mu2, sigma2 = np.mean(gen_features, axis=0), np.cov(gen_features, rowvar=False)
    
    # Compute FID
    diff = mu1 - mu2
    covmean = linalg.sqrtm(sigma1.dot(sigma2))
    if np.iscomplexobj(covmean):
        covmean = covmean.real
    fid = diff.dot(diff) + np.trace(sigma1 + sigma2 - 2 * covmean)
    return fid
