# Define a simple U-Net architecture
class UNet(nn.Module):
    def __init__(self):
        super(UNet, self).__init__()
        # Downsampling
        self.down1 = nn.Conv2d(3, 64, 3, padding=1)
        self.down2 = nn.Conv2d(64, 128, 3, padding=1, stride=2)  # Downsample
        # Bottleneck
        self.bottleneck = nn.Conv2d(128, 128, 3, padding=1)
        # Upsampling
        self.up1 = nn.ConvTranspose2d(128, 64, 3, stride=2, padding=1, output_padding=1)  # Upsample
        self.up2 = nn.Conv2d(64, 3, 3, padding=1)

    def forward(self, x, t):
        # t is the timestep (we'll ignore it in this simple version for now)
        x1 = F.relu(self.down1(x))
        x2 = F.relu(self.down2(x1))
        x3 = F.relu(self.bottleneck(x2))
        x4 = F.relu(self.up1(x3))
        return self.up2(x4)

# DDPM class
class DDPM(nn.Module):
    def __init__(self, num_timesteps=1000, beta_start=1e-4, beta_end=0.02):
        super(DDPM, self).__init__()
        self.num_timesteps = num_timesteps
        # Initialize tensors on CPU initially
        self.register_buffer('betas', torch.linspace(beta_start, beta_end, num_timesteps))
        self.register_buffer('alphas', 1.0 - self.betas)
        self.register_buffer('alpha_bars', torch.cumprod(self.alphas, dim=0))
        self.model = UNet()

    def forward_diffusion(self, x0, t, device):
        noise = torch.randn_like(x0).to(device)
        alpha_bar_t = self.alpha_bars[t].reshape(-1, 1, 1, 1).to(device)
        xt = torch.sqrt(alpha_bar_t) * x0 + torch.sqrt(1 - alpha_bar_t) * noise
        return xt, noise

    def train_step(self, x0, device):
        batch_size = x0.shape[0]
        t = torch.randint(0, self.num_timesteps, (batch_size,), device=device)
        xt, noise = self.forward_diffusion(x0, t, device)
        predicted_noise = self.model(xt, t)
        loss = F.mse_loss(predicted_noise, noise)
        return loss

    @torch.no_grad()
    def sample_one(self, device, img_size=(3, 64, 64)):
        x = torch.randn(1, *img_size).to(device)  # Generate one image at a time
        for t in reversed(range(self.num_timesteps)):
            t_tensor = torch.full((1,), t, device=device, dtype=torch.long)
            predicted_noise = self.model(x, t_tensor)
            alpha_t = self.alphas[t].to(device)
            alpha_bar_t = self.alpha_bars[t].to(device)
            x = (1 / torch.sqrt(alpha_t)) * (x - ((1 - alpha_t) / torch.sqrt(1 - alpha_bar_t)) * predicted_noise)
            if t > 0:
                x += torch.sqrt(self.betas[t]) * torch.randn_like(x).to(device)
        return x

    @torch.no_grad()
    def sample(self, device, n_samples=5, img_size=(3, 64, 64)):
        generated_images = []
        for _ in range(n_samples):
            img = self.sample_one(device, img_size)  # Generate one image at a time
            generated_images.append(img)
            clear_gpu_memory()  # Clear memory after each image
        return torch.cat(generated_images, dim=0)

# Early Stopping class
class EarlyStopping:
    def __init__(self, patience=3, delta=0, checkpoint_path="checkpoint_ddpm_yelp.pth"):
        """
        Args:
            patience (int): Number of epochs to wait for improvement before stopping.
            delta (float): Minimum change in loss to qualify as an improvement.
            checkpoint_path (str): Path to save the best model checkpoint.
        """
        self.patience = patience
        self.delta = delta
        self.checkpoint_path = checkpoint_path
        self.best_loss = float('inf')
        self.epochs_without_improvement = 0
        self.early_stop = False

    def __call__(self, loss, model):
        # Check if the current loss is better (lower) than the best loss
        if loss < self.best_loss - self.delta:
            self.best_loss = loss
            self.epochs_without_improvement = 0
            # Save the model checkpoint
            torch.save(model.state_dict(), self.checkpoint_path)
            print(f"New best loss: {self.best_loss:.6f}, saved checkpoint at {self.checkpoint_path}")
        else:
            self.epochs_without_improvement += 1
            print(f"No improvement in loss for {self.epochs_without_improvement}/{self.patience} epochs.")

        # Trigger early stopping if no improvement for 'patience' epochs
        if self.epochs_without_improvement >= self.patience:
            self.early_stop = True
            print("Early stopping triggered!")

# Initialize early stopping

# Training loop with gradient accumulation, early stopping, and checkpoints
optimizer = torch.optim.Adam(ddpm.parameters(), lr=1e-4)
num_epochs = 100  # Increased max epochs, but early stopping may stop earlier
accumulation_steps = 4  # Simulate a larger batch size
effective_batch_size = 8 * accumulation_steps  # 32

for epoch in range(num_epochs):
    ddpm.train()
    epoch_loss = 0
    batch_count = 0
    accumulated_loss = 0

    for batch in tqdm(dataloader, desc=f"Epoch {epoch+1}/{num_epochs}"):
        batch = batch.to(device)
        loss = ddpm.train_step(batch, device)
        loss = loss / accumulation_steps  # Normalize loss for accumulation
        loss.backward()

        accumulated_loss += loss.item()
        batch_count += 1

        if batch_count % accumulation_steps == 0:
            optimizer.step()
            optimizer.zero_grad()
            epoch_loss += accumulated_loss
            accumulated_loss = 0
            clear_gpu_memory()  # Clear memory after each optimization step

    # Handle remaining batches
    if batch_count % accumulation_steps != 0:
        optimizer.step()
        optimizer.zero_grad()
        epoch_loss += accumulated_loss

    # Compute average epoch loss
    avg_epoch_loss = epoch_loss / (len(dataloader) // accumulation_steps)
    print(f"Epoch {epoch+1}/{num_epochs}, Loss: {avg_epoch_loss:.6f}")

    # Check for early stopping
    early_stopping(avg_epoch_loss, ddpm)
    if early_stopping.early_stop:
        print(f"Stopping training at epoch {epoch+1} due to early stopping.")
        break

# Load the best model checkpoint

print("Best model loaded.")
