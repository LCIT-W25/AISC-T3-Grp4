    with torch.no_grad():

        latents = vae.encode(images).latent_dist.sample()

def decode_latents(latents):
    with torch.no_grad():
        latents = latents / 0.18215
        images = vae.decode(latents).sample
        images = (images.clamp(-1, 1) + 1) / 2  # Normalize to [0,1]
    return images

Define the U-Net Model for Latent Diffusion¶

The U-Net model in the Latent Diffusion Model is used for denoising the latent codes during the reverse diffusion process. We will use a simple U-Net architecture, as it works well for LDMs.

In [73]:
import torch
import torch.nn as nn
import torch.nn.functional as F

class UNet(nn.Module):
    def __init__(self, in_channels=4, out_channels=4, hidden_channels=64):
        super(UNet, self).__init__()

        self.enc1 = self.conv_block(in_channels, hidden_channels)
        self.enc2 = self.conv_block(hidden_channels, hidden_channels * 2)
        self.enc3 = self.conv_block(hidden_channels * 2, hidden_channels * 4)
        self.enc4 = self.conv_block(hidden_channels * 4, hidden_channels * 8)

        self.bottleneck = self.conv_block(hidden_channels * 8, hidden_channels * 16)

        self.dec4 = self.deconv_block(hidden_channels * 16, hidden_channels * 8)
        self.dec3 = self.deconv_block(hidden_channels * 8, hidden_channels * 4)
        self.dec2 = self.deconv_block(hidden_channels * 4, hidden_channels * 2)
        self.dec1 = self.deconv_block(hidden_channels * 2, hidden_channels)

        self.final_conv = nn.Conv2d(hidden_channels, out_channels, kernel_size=1)

    def conv_block(self, in_channels, out_channels):
        return nn.Sequential(
            nn.Conv2d(in_channels, out_channels, kernel_size=3, stride=1, padding=1),
            nn.ReLU(inplace=True),
            nn.Conv2d(out_channels, out_channels, kernel_size=3, stride=1, padding=1),
            nn.ReLU(inplace=True)
        )

    def deconv_block(self, in_channels, out_channels):
        return nn.Sequential(
            nn.ConvTranspose2d(in_channels, out_channels, kernel_size=4, stride=2, padding=1),
            nn.ReLU(inplace=True),
        )

    def forward(self, x):
        enc1 = self.enc1(x)
        enc2 = self.enc2(enc1)
        enc3 = self.enc3(enc2)
        enc4 = self.enc4(enc3)

        bottleneck = self.bottleneck(enc4)

        dec4 = self.dec4(bottleneck)
        dec4 = F.interpolate(dec4, size=enc4.shape[2:], mode="bilinear", align_corners=False)
        dec3 = self.dec3(dec4 + enc4)

        dec3 = F.interpolate(dec3, size=enc3.shape[2:], mode="bilinear", align_corners=False)
        dec2 = self.dec2(dec3 + enc3)

        dec2 = F.interpolate(dec2, size=enc2.shape[2:], mode="bilinear", align_corners=False)
        dec1 = self.dec1(dec2 + enc2)

        return self.final_conv(dec1)

Set Up the Diffusion Process¶

The diffusion process involves adding noise to the image during the forward pass and reversing this process to denoise during the reverse pass.

In [81]:
import torch
import torch.nn.functional as F

class DiffusionProcess:
    def __init__(self, beta_start=1e-4, beta_end=0.02, timesteps=500, device="mps"):
        self.timesteps = timesteps
        self.device = device
        self.beta_schedule = torch.linspace(beta_start, beta_end, timesteps, device=self.device)

    def forward_diffuse(self, x_0, t):
        noise = torch.randn_like(x_0, device=self.device)
        beta_t = self.beta_schedule[t].view(-1, 1, 1, 1)  # Expand dims to match x_0
        return torch.sqrt(1 - beta_t) * x_0 + torch.sqrt(beta_t) * noise

    def reverse_diffuse(self, model, x_t, t):
        """
        Reverse diffusion step: denoise the input latent code `x_t`.
        """
        model_output = model(x_t)

        # Ensure model_output matches x_t in spatial dimensions
        if model_output.shape != x_t.shape:
            model_output = F.interpolate(model_output, size=x_t.shape[2:], mode="bilinear", align_corners=False)
        
        return x_t - model_output

In the forward_diffuse function, we add noise to the image at each timestep.

The reverse_diffuse function uses the U-Net model to predict the denoised latent code.

def train_ldm(model, diffusion_process, dataloader, optimizer, num_epochs=10):
    model.train()
    for epoch in range(num_epochs):
        for images, _ in dataloader:
            optimizer.zero_grad()
            
            # Sample a random timestep
            t = torch.randint(0, diffusion_process.timesteps, (images.size(0),)).cuda()
            
            # Encode the image into latent space
            latents = vae.encode(images).latent_dist.sample()
            
            # Add noise to the latent code using the forward diffusion
            noisy_latents = diffusion_process.forward_diffuse(latents, t)
            
            # Perform reverse diffusion to denoise the latent code
            denoised_latents = diffusion_process.reverse_diffuse(model, noisy_latents, t)
            
            # Compute the loss (Mean Squared Error)
            loss = F.mse_loss(denoised_latents, latents)
            
            loss.backward()
            optimizer.step()

            print(f"Epoch [{epoch}/{num_epochs}], Loss: {loss.item()}")

def generate_images(model, diffusion_process, num_samples_per_label=1):
    """
    Generate images using the diffusion model with label conditioning.
    Ensures at least one image per label.
    """
    model.eval()
    generated_images = []
    image_labels = []

    with torch.no_grad():
        for label_idx in range(num_classes):  # Loop through each label
            for _ in range(num_samples_per_label):
                x_t = torch.randn(1, 4, 8, 8).to(device)  # Latent noise sample
                
                # Reverse diffusion process
                for t in reversed(range(diffusion_process.timesteps)):
                    x_t = diffusion_process.reverse_diffuse(model, x_t, t) 

                # Decode the latent code into an image
                decoded_image = vae.decode(x_t).sample.clamp(0, 1)  # Normalize to [0,1]

                generated_images.append(decoded_image)
                image_labels.append(reverse_label_map[label_idx])  # Convert index to label name

    return torch.cat(generated_images, dim=0), image_labels

def inception_score(images, splits=10):
    def get_inception_features(images):
        with torch.no_grad():
            images = F.interpolate(images, size=(299, 299), mode='bilinear', align_corners=False)
            features = inception_model(images)
            return F.softmax(features, dim=1)  # Convert logits to probabilities

    scores = []
    for split in range(splits):
        part = images[split * (len(images) // splits):(split + 1) * (len(images) // splits)].to(device)
        prob = get_inception_features(part)
        mean_prob = prob.mean(dim=0, keepdim=True)
        
        kl_div = (prob * (prob.log() - mean_prob.log())).sum(dim=1).mean()
        scores.append(torch.exp(kl_div).item())

        # Free up memory
        del part, prob
        torch.mps.empty_cache()

    return sum(scores) / len(scores)

# Function to extract features using InceptionV3
def get_inception_features(images):
    """Extract features from InceptionV3."""
    with torch.no_grad():
        features = inception_model(images)
        if isinstance(features, tuple):  
            features = features[0]  # Ignore aux_logits output
        return features.view(features.size(0), -1)

# Function to calculate Fréchet Inception Distance (FID)
def calculate_fid(real_images, generated_images):
    """Compute Fréchet Inception Distance (FID)."""
    real_images, generated_images = real_images.to(device), generated_images.to(device)

    # Extract features
    real_features = get_inception_features(real_images)
    generated_features = get_inception_features(generated_images)

    # Move to CPU for NumPy calculations
    real_features, generated_features = real_features.cpu().numpy(), generated_features.cpu().numpy()

    # Compute mean and covariance
    mu_real, sigma_real = real_features.mean(axis=0), np.cov(real_features, rowvar=False)
    mu_gen, sigma_gen = generated_features.mean(axis=0), np.cov(generated_features, rowvar=False)

    # Compute sqrtm of covariance product
    covmean = sqrtm(sigma_real @ sigma_gen)
    
    # Numerical stability: Ensure it's real
    if np.iscomplexobj(covmean):
        covmean = covmean.real

    # Compute FID
    fid_score = np.sum((mu_real - mu_gen) ** 2) + np.trace(sigma_real + sigma_gen - 2 * covmean)
    return fid_score

The FID score is 3133.83, seems quite high. Generally, FID scores range from 0 (perfect match) to higher values, with lower scores indicating better similarity between the generated and real images.

A high FID score often suggests that there is a significant difference between the real and generated images. This can happen for various reasons

Quality of Generated Images: The images generated might not be realistic enough, which could explain the high FID score.

Data Mismatch: The real images used for evaluation may not match the type of images generated by the model, or the preprocessing steps might not align.

Insufficient Training: If the generative model isn't trained enough, its generated samples may not resemble the real data closely, leading to a high FID score.
