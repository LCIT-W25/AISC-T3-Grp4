import pandas as pd
import numpy as np
import os
from PIL import Image

# Define paths
json_path = r"D:\Loyalist college\computer vision\Yelp-Photos\Yelp Photos\yelp_photos\photos.json"
image_dir = r"D:\Loyalist college\computer vision\Yelp-Photos\Yelp Photos\yelp_photos\photos"

# Load the JSON file
df = pd.read_json(json_path, lines=True)

# Select only 10,000 random images 
df = df.sample(n=10000, random_state=42)

# Create full image paths using the correct folder
df['image_path'] = df['photo_id'].apply(lambda x: os.path.join(image_dir, f"{x}.jpg"))

# Check if files exist
df['exists'] = df['image_path'].apply(os.path.exists)

# Show missing files
missing_files = df[~df['exists']]
if not missing_files.empty:
    print("\n Warning: Some image files are missing! Here are the first 5 missing files:")
    display(missing_files[['photo_id', 'image_path']].head())
else:
    print("\n All image paths are valid!")

df.drop(columns=['exists'], inplace=True)

# Define preprocessing function
IMG_SIZE = (32, 32) 

def preprocess_image(image_path):
    """Loads an image, converts to RGB, resizes to (32x32), and flattens it."""
    try:
        img = Image.open(image_path).convert('RGB')  # Ensure RGB mode
        img = img.resize(IMG_SIZE)  # Resize
        return np.array(img).flatten().astype(np.float16)  # Convert to float16 to save memory
    except Exception as e:
        print(f" Error processing image {image_path}: {e}")
        return None  # Skip failed images

df['image_array'] = df['image_path'].apply(preprocess_image)

# Drop rows where images could not be processed
df.dropna(subset=['image_array'], inplace=True)

# Print results
print("\n Preprocessing completed successfully!")
print(f"Total valid images processed: {len(df)}")

# Display first 5 processed images (flattened arrays)
df[['photo_id', 'image_array']].head()

metrics = ['euclidean', 'manhattan'] 
weights = ['uniform', 'distance']  

# Create a dictionary to store results
results = {}

# Train and evaluate the models with different configurations
for k in k_values:
    for metric in metrics:
        for weight in weights:
            print(f"Training kNN with k={k}, metric={metric}, weight={weight}")
            knn = KNeighborsClassifier(n_neighbors=k, metric=metric, weights=weight)
            knn.fit(X_train, y_train)
            
            # Calculate accuracy
            train_acc = knn.score(X_train, y_train)
            test_acc = knn.score(X_test, y_test)
            
            # Store the results
            results[(k, metric, weight)] = {
                'train_accuracy': train_acc,
                'test_accuracy': test_acc
            }

# Display the results
for config, scores in results.items():
    print(f"\nConfig: {config}")
    print(f"Train Accuracy: {scores['train_accuracy']:.4f}, Test Accuracy: {scores['test_accuracy']:.4f}")
