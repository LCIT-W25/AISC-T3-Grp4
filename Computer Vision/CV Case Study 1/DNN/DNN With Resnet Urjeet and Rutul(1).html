IMG_SIZE = (64, 64)

def preprocess_image(image_path):
    try:
        img = Image.open(image_path).convert('RGB')
        img = img.resize(IMG_SIZE)
        return np.array(img)  
    except Exception as e:
        return None

from tensorflow.keras.applications.resnet50 import preprocess_input

# ResNet50 expects images of size 224x224
IMG_SIZE_RESNET = (224, 224)

def resize_images(images):
    # Resize and convert images to uint8
    resized_images = np.array([
        np.array(Image.fromarray((img * 255).astype('uint8')).resize(IMG_SIZE_RESNET)) 
        for img in images
    ])
    
    # Apply ResNet-specific preprocessing
    return preprocess_input(resized_images)

# Resize and preprocess the training and testing data
X_train_resized = resize_images(X_train)
X_test_resized = resize_images(X_test)

)

# Function to preprocess the image
def preprocess_image_for_resnet(image_path):
    IMG_SIZE_RESNET = (224, 224)
    try:
        img = Image.open(image_path).convert('RGB')
        img = img.resize(IMG_SIZE_RESNET)
        img_array = np.array(img)
        img_array = preprocess_input(img_array)  # ResNet50 preprocessing
        return img_array
    except Exception as e:
        print(f"❌ Error loading image: {e}")
        return None

# Function to predict and display both predicted and actual labels
def predict_image(photo_id):
    # Construct the image path
    image_path = os.path.join(image_dir, f"{photo_id}.jpg")

    # Preprocess the image
    image_array = preprocess_image_for_resnet(image_path)
    
    if image_array is not None:
        # Add batch dimension for prediction
        image_input = np.expand_dims(image_array, axis=0)

        # Predict probabilities and get the predicted class
        predicted_probabilities = model_resnet.predict(image_input)
        predicted_class = np.argmax(predicted_probabilities)
        predicted_label = label_encoder.inverse_transform([predicted_class])[0]

        # Retrieve the actual label from the DataFrame (if exists)
        actual_label = "Unknown"  # Default value
        if photo_id in df['photo_id'].values:
            actual_label = df.loc[df['photo_id'] == photo_id, 'label'].values[0]

        # Display the image with both predicted and actual labels
        plt.figure(figsize=(4, 4))
        plt.imshow(Image.open(image_path))
        plt.title(f"Predicted: {predicted_label}\nActual: {actual_label}", fontsize=14)
        plt.axis('off')
        plt.show()

        # Print the labels in the console
        print(f"✅ Predicted Label: {predicted_label}")
        print(f"🎯 Actual Label: {actual_label}")
    else:
        print(f"❌ Unable to load image for photo_id: {photo_id}")
