#  Set image size for VGG model
IMG_SIZE = (224, 224)

#  Function to preprocess images
def load_and_preprocess_image(image_path):
    try:
        img = cv2.imread(image_path)  # Load image
        if img is None:
            return None  # Skip missing/corrupt images
        img = cv2.resize(img, IMG_SIZE)  # Resize to 224x224
        img = img / 255.0  # Normalize pixel values
        return img
    except Exception as e:
        print(f"⚠️ Error processing image: {image_path} - {e}")
        return None

#  Apply image preprocessing
df['processed_image'] = df['image_path'].apply(load_and_preprocess_image)

#  Drop any rows with missing processed images
df = df.dropna(subset=['processed_image']).reset_index(drop=True)

#  Convert processed images to NumPy array
X_images = np.array(df['processed_image'].tolist(), dtype=np.float32)

print(f"\n Successfully processed {len(df)} images.")

#  Load VGG16 as base model (without the top classification layer)
base_model = VGG16(weights="imagenet", include_top=False, input_shape=(224, 224, 3))

#  Freeze the convolutional layers (so they don’t get retrained initially)
base_model.trainable = False

#  Image Input & Feature Extraction
image_input = Input(shape=(224, 224, 3), name="image_input")
x = base_model(image_input, training=False)  # Pass input through VGG16
x = GlobalAveragePooling2D()(x)  # Use GlobalAveragePooling instead of Flatten (better generalization)
x = Dense(512, activation="relu")(x)  # Fully connected layer
x = Dropout(0.5)(x)  # Dropout for regularization
x = BatchNormalization()(x)  # Normalize activations
image_features = Dense(256, activation="relu")(x)  # Final image feature layer

#  Metadata Input Processing
metadata_input = Input(shape=(3,), name="metadata_input")  # 3 features: stars, review_count, category
y = Dense(64, activation="relu")(metadata_input)  # Process metadata
y = BatchNormalization()(y)
y = Dense(32, activation="relu")(y)

#  Merge Image & Metadata Features
merged = Concatenate()([image_features, y])  # Merge both inputs
merged = Dense(128, activation="relu")(merged)  # Learn joint representation
merged = Dropout(0.5)(merged)  # Regularization
merged = BatchNormalization()(merged)  # Normalize after merging

#  Output Layer (Multi-class classification)
output = Dense(y_train.shape[1], activation="softmax", name="output_layer")(merged)

#  Define the Model
model = Model(inputs=[image_input, metadata_input], outputs=output)

#  Compile the Model with Lower Learning Rate
model.compile(optimizer="adam", loss="categorical_crossentropy", metrics=["accuracy"])

#  Print Model Summary
model.summary()
