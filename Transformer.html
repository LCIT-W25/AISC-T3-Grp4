Possible Model Improvements¶

  • Increase Embedding Size (128 → 300) → Better semantic understanding (+0.5% to +1% accuracy).
  • More Transformer Layers (6 → 12), More Heads (8 → 16) → Improves context learning (+1% to +2% accuracy).
  • Use Pretrained Embeddings (GloVe, FastText) → Faster convergence, better generalization (+1% to +3% AUC).
  • Better LR Scheduling (One-Cycle LR instead of Cosine) → Faster, more stable training (-5% to -10% training time).
  • Regularization Tuning (Dropout 0.2 → Dynamic 0.1-0.4) → Reduces overfitting (+0.5% to +1.2% macro AUC).
  • Larger Batch with Mixed Precision (64 → 128 FP16) → 2x faster training, no accuracy loss.

import numpy as np

# Function to generate sinusoidal position embeddings
def get_sinusoidal_embeddings(max_len, embedding_dim):
    position = np.arange(max_len)[:, np.newaxis]  # Positions (0 to max_len-1)
    div_term = np.exp(np.arange(0, embedding_dim, 2) * -(np.log(10000.0) / embedding_dim))

    # Compute sin for even indices, cos for odd indices
    pos_embedding = np.zeros((max_len, embedding_dim))
    pos_embedding[:, 0::2] = np.sin(position * div_term)
    pos_embedding[:, 1::2] = np.cos(position * div_term)

    return pos_embedding

import torch.nn as nn

class InitialTransformer(nn.Module):
    def __init__(self, vocab_size, embedding_dim, max_seq_length, num_classes=5):
        super(InitialTransformer, self).__init__()

        # Word Embedding Layer
        self.word_embedding = nn.Embedding(vocab_size, embedding_dim)

        # Transformer Encoder Layer (Simpler Version)
        self.transformer_layer = nn.TransformerEncoderLayer(
            d_model=embedding_dim,
            nhead=4,  # Fewer attention heads (simpler)
            dim_feedforward=512,  # Smaller feedforward network
            dropout=0.1,  # Lower dropout
            batch_first=True
        )

        # Single Transformer Encoder Layer
        self.transformer_encoder = nn.TransformerEncoder(
            self.transformer_layer, num_layers=3  # Fewer layers
        )

        # Output layer (Classification)
        self.fc = nn.Linear(embedding_dim, num_classes)

    def forward(self, x):
        # Word embeddings
        word_emb = self.word_embedding(x)

        # Transformer processing
        x = self.transformer_encoder(word_emb)

        # Classification layer (using first token representation)
        x = self.fc(x[:, 0, :])

        return x

# Initialize Initial Model

num_epochs = 10  # Number of epochs for training
best_val_loss = float('inf')  # Track the best validation loss
patience = 3  # Early stopping patience
patience_counter = 0

for epoch in range(num_epochs):
    # Training phase
    initial_model.train()
    total_train_loss = 0
    train_correct = 0
    train_total = 0
    
    for batch_idx, (texts, labels) in enumerate(tqdm(train_loader, desc=f'Epoch {epoch+1}/{num_epochs}')):
        texts, labels = texts.to(device), labels.to(device)
        
        optimizer.zero_grad()
        outputs = initial_model(texts)
        loss = criterion(outputs, labels)
        loss.backward()
        
        # Gradient clipping to prevent exploding gradients
        torch.nn.utils.clip_grad_norm_(initial_model.parameters(), max_norm=1.0)
        
        optimizer.step()
        
        total_train_loss += loss.item()
        _, predicted = torch.max(outputs.data, 1)
        train_total += labels.size(0)
        train_correct += (predicted == labels).sum().item()
    
    avg_train_loss = total_train_loss / len(train_loader)
    train_accuracy = 100 * train_correct / train_total

    # Validation phase
    initial_model.eval()
    total_val_loss = 0
    val_correct = 0
    val_total = 0
    
    with torch.no_grad():
        for texts, labels in val_loader:
            texts, labels = texts.to(device), labels.to(device)
            outputs = initial_model(texts)
            loss = criterion(outputs, labels)
            
            total_val_loss += loss.item()
            _, predicted = torch.max(outputs.data, 1)
            val_total += labels.size(0)
            val_correct += (predicted == labels).sum().item()
    
    avg_val_loss = total_val_loss / len(val_loader)
    val_accuracy = 100 * val_correct / val_total

    print(f"\n✅ Epoch [{epoch+1}/{num_epochs}]")
    print(f"Train Loss: {avg_train_loss:.4f}, Train Accuracy: {train_accuracy:.2f}%")
    print(f"Val Loss: {avg_val_loss:.4f}, Val Accuracy: {val_accuracy:.2f}%")

    # Learning rate scheduling
    scheduler.step()
    
    # Early stopping check
    if avg_val_loss < best_val_loss:
        best_val_loss = avg_val_loss
        patience_counter = 0
        torch.save(initial_model.state_dict(), "initial_transformer_model.pth")  # Save best model
    else:
        patience_counter += 1
        if patience_counter >= patience:
            print(f"🚨 Early stopping triggered after {epoch+1} epochs")
            break
