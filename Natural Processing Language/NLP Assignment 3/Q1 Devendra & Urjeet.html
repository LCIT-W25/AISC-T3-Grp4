# Add positional encoding (for Causal Transformer)
def get_positional_encoding(max_len, embedding_dim):
    position = np.arange(max_len)[:, np.newaxis]
    div_term = np.exp(np.arange(0, embedding_dim, 2) * -(np.log(10000.0) / embedding_dim))
    pos_encoding = np.zeros((max_len, embedding_dim))
    pos_encoding[:, 0::2] = np.sin(position * div_term)
    pos_encoding[:, 1::2] = np.cos(position * div_term)
    return tf.constant(pos_encoding, dtype=tf.float32)

pos_encoding = get_positional_encoding(max_len, embedding_dim)
embedding_layer = embedding_layer + pos_encoding

# Add the Transformer block (Causal Masked Attention)
for _ in range(num_layers):
    attention = layers.MultiHeadAttention(num_heads=num_heads, key_dim=embedding_dim)(embedding_layer, embedding_layer)
    attention = layers.Dropout(dropout_rate)(attention)
    attention = layers.LayerNormalization(epsilon=1e-6)(embedding_layer + attention)  # Residual connection

    # Feed-forward network
    ffn = layers.Dense(embedding_dim * 4, activation='relu')(attention)
    ffn = layers.Dense(embedding_dim)(ffn)
    ffn = layers.Dropout(dropout_rate)(ffn)
    ffn = layers.LayerNormalization(epsilon=1e-6)(attention + ffn)  # Residual connection
    
    embedding_layer = ffn  # Output of this layer is input to the next

class Seq2SeqDataGenerator(tf.keras.utils.Sequence):
    def __init__(self, encoder_inputs, decoder_targets, batch_size, **kwargs):
        super().__init__(**kwargs)  # <-- This line fixes the warning
        self.encoder_inputs = encoder_inputs
        self.decoder_targets = decoder_targets
        self.batch_size = batch_size

    def __len__(self):
        return math.ceil(len(self.encoder_inputs) / self.batch_size)

    def __getitem__(self, idx):
        batch_encoder_inputs = self.encoder_inputs[idx * self.batch_size:(idx + 1) * self.batch_size]
        batch_decoder_targets = self.decoder_targets[idx * self.batch_size:(idx + 1) * self.batch_size]
        return batch_encoder_inputs, batch_decoder_targets

# 3. ROUGE-L Evaluation
def rouge_l_eval(predicted, actual):
    scorer = rouge_scorer.RougeScorer(['rougeL'], use_stemmer=True)
    scores = [scorer.score(' '.join(a), ' '.join(p))['rougeL'].fmeasure for p, a in zip(predicted, actual)]
    return np.mean(scores)

# === POSITIONAL ENCODING ===
def get_positional_encoding(max_len, embedding_dim):
    position = np.arange(max_len)[:, np.newaxis]
    div_term = np.exp(np.arange(0, embedding_dim, 2) * -(np.log(10000.0) / embedding_dim))
    pos_encoding = np.zeros((max_len, embedding_dim))
    pos_encoding[:, 0::2] = np.sin(position * div_term)
    pos_encoding[:, 1::2] = np.cos(position * div_term)
    return tf.constant(pos_encoding, dtype=tf.float32)

pos_encoding = get_positional_encoding(MAX_LEN, embedding_dim)
embedding_layer = embedding_layer + pos_encoding

# === TRANSFORMER BLOCKS ===
x = embedding_layer
for _ in range(num_layers):
    attn = tf.keras.layers.MultiHeadAttention(num_heads=num_heads, key_dim=embedding_dim)(x, x)
    attn = tf.keras.layers.Dropout(dropout_rate)(attn)
    x = tf.keras.layers.LayerNormalization(epsilon=1e-6)(x + attn)
    ffn = tf.keras.layers.Dense(embedding_dim * 4, activation='relu')(x)
    ffn = tf.keras.layers.Dense(embedding_dim)(ffn)
    ffn = tf.keras.layers.Dropout(dropout_rate)(ffn)
    x = tf.keras.layers.LayerNormalization(epsilon=1e-6)(x + ffn)

# === OUTPUT LAYER ===

class Seq2SeqDataGenerator(tf.keras.utils.Sequence):
    def __init__(self, encoder_inputs, decoder_targets, batch_size):
        self.encoder_inputs = encoder_inputs
        self.decoder_targets = decoder_targets
        self.batch_size = batch_size

    def __len__(self):
        return math.ceil(len(self.encoder_inputs) / self.batch_size)

    def __getitem__(self, idx):
        batch_encoder_inputs = self.encoder_inputs[idx * self.batch_size:(idx + 1) * self.batch_size]
        batch_decoder_targets = self.decoder_targets[idx * self.batch_size:(idx + 1) * self.batch_size]

from sklearn.metrics import roc_auc_score, roc_curve
import matplotlib.pyplot as plt
import numpy as np

# === SAMPLE BATCH ===
sample_inputs, sample_targets = val_generator[0]
predictions = model.predict(sample_inputs)  # (batch_size, MAX_LEN, vocab_size)

# === Flatten to (num_tokens, vocab_size) and (num_tokens,)
flat_preds = predictions.reshape(-1, predictions.shape[-1])  # (batch_size * MAX_LEN, vocab_size)
flat_labels = sample_targets.flatten()  # (batch_size * MAX_LEN,)

# === Top-K Most Frequent Token IDs in Labels ===
unique_ids, counts = np.unique(flat_labels, return_counts=True)
top_k_ids = unique_ids[np.argsort(-counts)][:5]

# === Plot ROC Curves ===
plt.figure(figsize=(10, 6))
valid_auc = []

for token_id in top_k_ids:
    y_true = (flat_labels == token_id).astype(int)
    y_scores = flat_preds[:, token_id]

    if len(np.unique(y_true)) < 2:
        print(f"⚠️ Skipping Token ID {token_id} — only one class in labels.")
        continue

    auc = roc_auc_score(y_true, y_scores)
    fpr, tpr, _ = roc_curve(y_true, y_scores)
    plt.plot(fpr, tpr, label=f"Token {token_id} (AUC = {auc:.2f})")
    valid_auc.append((token_id, auc))

plt.plot([0, 1], [0, 1], linestyle='--', color='gray')
plt.xlabel("False Positive Rate")
plt.ylabel("True Positive Rate")
plt.title("AUC for Top 5 Token Classes (All Positions)")
plt.legend()
plt.grid(True)
plt.tight_layout()
plt.show()

# === Summary ===
if valid_auc:
    print("\n✅ AUC Scores Summary:")
    for tid, auc in valid_auc:
        print(f"Token ID {tid}: AUC = {auc:.4f}")
else:
    print("⚠️ Still no valid AUC scores — need more varied targets.")
